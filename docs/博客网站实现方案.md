系统概述
设计原则

渐进式开发: 从MVP到完整系统,分阶段实施
模块化架构: 每个服务独立部署,可插拔设计
无感知扩展: 新功能不影响现有系统
性能优先: 边缘计算 + 多层缓存
用户体验: 公开访问为主,交互时引导登录

技术选型
前端框架: React 18 + Vite
UI组件库: shadcn/ui + Tailwind CSS
状态管理: Zustand (轻量级)
路由管理: React Router v6
Markdown: react-markdown + remark/rehype

后端运行时: Cloudflare Workers
API框架: Hono (轻量级,性能优秀)
认证方案: JWT + OAuth 2.0

数据存储:
  - D1 (SQLite): 关系型数据
  - KV: 缓存 + 会话
  - R2: 对象存储
  - Analytics Engine: 数据分析
  - Durable Objects: 实时功能 (v3+)
  - Queue: 异步任务 (v2+)

开发工具:
  - Wrangler CLI: 部署管理
  - TypeScript: 类型安全
  - Vitest: 单元测试
  - Playwright: E2E测试

CREATE TABLE users (

  email TEXT UNIQUE NOT NULL,
  username TEXT UNIQUE NOT NULL,
  password_hash TEXT,
  display_name TEXT,
  avatar_url TEXT,
  bio TEXT,
  role TEXT DEFAULT 'user', -- user, admin
  oauth_provider TEXT, -- github, google
  oauth_id TEXT,
  email_verified BOOLEAN DEFAULT 0,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  updated_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- 会话表
CREATE TABLE sessions (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  user_id INTEGER NOT NULL,
  token TEXT UNIQUE NOT NULL,
  expires_at DATETIME NOT NULL,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (user_id) REFERENCES users(id)
);

-- 文章表
CREATE TABLE posts (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  author_id INTEGER NOT NULL,
  title TEXT NOT NULL,
  slug TEXT UNIQUE NOT NULL,
  content TEXT NOT NULL,
  excerpt TEXT,
  cover_image TEXT,
  status TEXT DEFAULT 'draft', -- draft, published, archived
  published_at DATETIME,
  view_count INTEGER DEFAULT 0,
  reading_time INTEGER DEFAULT 0, -- 分钟
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  updated_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (author_id) REFERENCES users(id)
);

-- 分类表
CREATE TABLE categories (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  name TEXT NOT NULL,
  slug TEXT UNIQUE NOT NULL,
  description TEXT,
  icon TEXT,
  sort_order INTEGER DEFAULT 0,
  post_count INTEGER DEFAULT 0,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- 文章分类关联
CREATE TABLE post_categories (
  post_id INTEGER NOT NULL,
  category_id INTEGER NOT NULL,
  PRIMARY KEY (post_id, category_id),
  FOREIGN KEY (post_id) REFERENCES posts(id) ON DELETE CASCADE,
  FOREIGN KEY (category_id) REFERENCES categories(id) ON DELETE CASCADE
);

-- 标签表
CREATE TABLE tags (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  name TEXT NOT NULL,
  slug TEXT UNIQUE NOT NULL,
  post_count INTEGER DEFAULT 0,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- 文章标签关联
CREATE TABLE post_tags (
  post_id INTEGER NOT NULL,
  tag_id INTEGER NOT NULL,
  PRIMARY KEY (post_id, tag_id),
  FOREIGN KEY (post_id) REFERENCES posts(id) ON DELETE CASCADE,
  FOREIGN KEY (tag_id) REFERENCES tags(id) ON DELETE CASCADE
);

-- 评论表
CREATE TABLE comments (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  post_id INTEGER NOT NULL,
  user_id INTEGER NOT NULL,
  parent_id INTEGER, -- 父评论ID,用于嵌套回复
  content TEXT NOT NULL,
  status TEXT DEFAULT 'approved', -- pending, approved, rejected, deleted
  like_count INTEGER DEFAULT 0,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  updated_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (post_id) REFERENCES posts(id) ON DELETE CASCADE,
  FOREIGN KEY (user_id) REFERENCES users(id),
  FOREIGN KEY (parent_id) REFERENCES comments(id) ON DELETE CASCADE
);

-- 媒体文件表
CREATE TABLE media (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  user_id INTEGER NOT NULL,
  filename TEXT NOT NULL,
  original_name TEXT,
  mime_type TEXT,
  size INTEGER, -- 字节
  width INTEGER,
  height INTEGER,
  r2_key TEXT NOT NULL, -- R2存储的key
  url TEXT NOT NULL,
  thumbnail_url TEXT,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (user_id) REFERENCES users(id)
);

CREATE INDEX idx_comments_parent ON comments(parent_id);
CREATE INDEX idx_comments_status ON comments(status);

### 🔄 服务降级与回滚
```javascript
// 服务降级示例
class ServiceFallback {
  constructor(primaryService, fallbackService) {
    this.primary = primaryService;
    this.fallback = fallbackService;
    this.circuitBreaker = new CircuitBreaker();
  }
  
  async call(method, ...args) {
    // 如果熔断器开启,直接使用降级服务
    if (this.circuitBreaker.isOpen()) {
      return this.fallback[method](...args);
    }
    
    try {
      const result = await this.primary[method](...args);
      this.circuitBreaker.recordSuccess();
      return result;
    } catch (error) {
      this.circuitBreaker.recordFailure();
      
      // 使用降级服务
      if (this.fallback) {
        console.warn(`Falling back for ${method}:`, error);
        return this.fallback[method](...args);
      }
      
      throw error;
    }
  }
}

// 使用示例
const searchService = new ServiceFallback(
  env.SEARCH_SERVICE, // Algolia高级搜索
  {
    // 降级为简单SQL搜索
    search: async (query) => {
      return await env.DB.prepare(
        'SELECT * FROM posts WHERE title LIKE ?'
      ).bind(`%${query}%`).all();
    }
```
